cURL Examples

# GET all
curl http://localhost:3000/movies

# GET by id
curl http://localhost:3000/movies/1

# POST
curl -X POST http://localhost:3000/movies \
  -H "Content-Type: application/json" \
  -d '{"title":"Tenet","director":"Christopher Nolan","year":2020}'

# PUT
curl -X PUT http://localhost:3000/movies/2 \
  -H "Content-Type: application/json" \
  -d '{"title":"Updated","director":"New","year":2015}'

# DELETE
curl -X DELETE http://localhost:3000/movies/3
